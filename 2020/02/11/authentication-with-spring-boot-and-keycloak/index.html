package com.javachen.controller;

import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.web.bind.annotation.GetMapping;
import org.springframework.web.bind.annotation.RestController;

import javax.servlet.ServletException;
import javax.servlet.http.HttpServletRequest;

@RestController
public class ApplicationController {
    @Autowired
    private HttpServletRequest request;

    @GetMapping(value = "/logout")
    public String logout() throws ServletException {
        request.logout();
        return "logout success";
    }

    @GetMapping(value = "/api/user")
    public String user() {
        return "User";
    }

    @GetMapping(value = "/api/admin")
    public String admin() {
        return "Admin";
    }
}

package com.javachen.config;

import org.keycloak.adapters.KeycloakDeployment;
import org.keycloak.adapters.KeycloakDeploymentBuilder;
import org.keycloak.adapters.spi.HttpFacade;
import org.keycloak.adapters.springboot.KeycloakSpringBootConfigResolver;
import org.keycloak.adapters.springboot.KeycloakSpringBootProperties;
import org.springframework.context.annotation.Configuration;

@Configuration
public class CustomKeycloakSpringBootConfigResolver extends KeycloakSpringBootConfigResolver {
    private final KeycloakDeployment keycloakDeployment;

    public CustomKeycloakSpringBootConfigResolver(KeycloakSpringBootProperties properties) {
        keycloakDeployment = KeycloakDeploymentBuilder.build(properties);
    }

    @Override
    public KeycloakDeployment resolve(HttpFacade.Request facade) {
        return keycloakDeployment;
    }
}

测试

浏览器测试

启动应用之后，通过浏览器访问：http://localhost:8081/api/user ，会看到：

1

User

浏览器访问：http://localhost:8081/api/admin ，会看到：

1

Admin

说明没有设置权限约束，接下来修改配置文件，添加：

1
2
3

keycloak.securityConstraints[0].authRoles[0]=*
keycloak.securityConstraints[0].securityCollections[0].name=protected
keycloak.securityConstraints[0].securityCollections[0].patterns[0]=/api/*
  • 上面配置意思是：拥有任意角色的用户才能访问 /api/*

重启应用，通过浏览器访问：http://localhost:8081/api/user ，会跳转到 keycloak sso 登陆页面：

说明：必须登录Keycloak才能访问。

输入用户名user和秘密123456，会返回如下内容：

1

User

这时候，访问 http://localhost:8081/api/admin ，会返回如下内容：

1

Admin

修改配置文件改为：

1
2
3
4
5
6
7

keycloak.securityConstraints[0].authRoles[0]=user
keycloak.securityConstraints[0].securityCollections[0].name=protected
keycloak.securityConstraints[0].securityCollections[0].patterns[0]=/api/user

keycloak.securityConstraints[1].authRoles[0]=admin
keycloak.securityConstraints[1].securityCollections[0].name=protected
keycloak.securityConstraints[1].securityCollections[0].patterns[0]=/api/admin
  • 说明：上面配置的是只有user角色访问/api/user、admin角色访问/api/admin

重启应用，通过浏览器访问 http://localhost:8081/logout ，退出当前session，然后通过浏览器访问：http://localhost:8081/api/user ，会跳转到 keycloak sso 登陆页面，使用user用户登录，返回

命令行测试

如果设置下面参数，则可以使用命令行访问，而不用跳转到登陆页面：

1

keycloak.bearer-only=true

使用user用户获取token：

1

export access_token=`curl -ss --data "grant_type=password&client_id=app-springboot-rest&client_secret=secret&username=user&password=123456" http://localhost:8080/auth/realms/demo/protocol/openid-connect/token | jq -r .access_token`

访问api：

1
2
3
4
5

curl -H "Authorization: bearer $access_token" http://localhost:8081/api/user && echo

curl -H "Authorization: bearer $access_token" http://localhost:8081/api/admin && echo

curl -H "Authorization: bearer $access_token" http://localhost:8081/logout && echo
